\section{OBJETIVOS}
	\begin{itemize}
		\item[$\bullet$] Comprovar na prática a validade e limitações de um
		misturador e um modulador balanceado;
		\item[$\bullet$] Determinar o nível de supressão da portadora à saída
		do modulador balanceado baseado em circuito integrad;
		\item[$\bullet$] Avaliar o conteúdo harmônico de um misturador e um
		modulador a partir do analisador de espectro;
\end{itemize}
